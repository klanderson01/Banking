function parseCSV(csvContent, bankFormat) {
    const lines = csvContent.split('\n');
    const headers = lines[0].split(',');
    
    return lines.slice(1).map(line => {
        const values = parseCSVLine(line);
        return mapTransaction(values, headers, bankFormat);
    }).filter(transaction => transaction.amount);
}

function autoCategorizTransaction(description) {
    const patterns = {
        'Housing': ['rent', 'mortgage'],
        'Transportation': ['gas', 'fuel', 'car'],
        'Food': ['grocery', 'restaurant']
    };
    
    for (const [category, keywords] of Object.entries(patterns)) {
        if (keywords.some(keyword => 
            description.toLowerCase().includes(keyword))) {
            return category;
        }
    }
    return 'Uncategorized';
}
